\title{Any-Order Adaptive Editing for Masked Diffusion LLMs}
\author{}
\date{}

% ======================================================================
\begin{abstract}
% ======================================================================
Masked diffusion language models (MDLMs) decode in parallel but pay uniform compute per step, regardless of which positions are easy and which are hard.
We propose \textbf{Any-Order Adaptive Editing (AOAE)}, a self-speculative framework for MoE MDLMs with three contributions.
\emph{(i)~A drafter--verifier split} of a single frozen model: hard routing produces a cheap drafter (${\sim}$1.4B active), normal routing produces the verifier; a composition knob $\gamma$ tunes speed against quality and recovers the verifier distribution exactly at $\gamma{=}0$.
\emph{(ii)~A two-cache mechanism} cleanly separating two notions a single cache conflates: a transient frontier $\cK_{\text{spec}}$ of unverified draft tokens (verification debt, cleared at every verifier event) and a persistent stability cache $\cK_{\text{stable}}$ of positions predicted to remain reusable across multiple steps (evicted only on remask or rejection).
\emph{(iii)~A learned plug-in caching head}: a lightweight policy (${\sim}$500K parameters) trained end-to-end with GRPO predicts $\cK_{\text{stable}}$ membership and refresh access from the verifier's own state.
\end{abstract}

% ======================================================================
\section{Background}
\label{sec:background}
% ======================================================================

Let $\bx_0 = (x_0^1, \dots, x_0^L) \in \cV^L$ be a token sequence with mask token $\mask$.
The forward process corrupts each token independently:
\begin{equation}
    q(x_t^k \mid x_0^k) = \begin{cases} 1{-}t, & x_t^k = x_0^k, \\ t, & x_t^k = \mask, \end{cases}
    \quad t \in [0,1].
    \label{eq:forward}
\end{equation}
A denoising network $p_{\btheta}$ trained via the masked ELBO~\cite{ou2024your,sahoo2024simple} starts from $\by_T = (\mask,\ldots,\mask)$ and iteratively selects $\cU_t \subseteq \cM_t$ to unmask; this choice governs both speed and quality~\cite{jazbec2025learning,wu2025fastdllm}.
Prior accelerations split along three axes: adaptive unmasking schedules~\cite{jazbec2025learning,kim2025trainworst}, KV-cache reuse restricted to block or prefix patterns~\cite{ma2025dkv,elastic_cache}, and confidence-based self-correction~\cite{kim2025prism}.
AOAE unifies these into a single learned policy guided by a self-speculative verifier.

\section{Method}
\label{sec:method}
% ======================================================================

\subsection{Dual-Model MoE Architecture}

LLaDA~2.1-mini~\cite{llada21} is a 16B-parameter MoE with $E{=}128$ routed experts and native top-$k{=}8$ (${\sim}$1.4B active).
AOAE runs two modes of the \emph{same} frozen model, sharing weights and toggling routing in place:

\textbf{Drafter (hard-routed).}  Native top-$k$, paired with an aggressive inference shortcut (S-mode thresholding, prefix-KV reuse on $\cC_{\text{aux}}$).  Each unmasked token is pushed to $\cK_{\text{spec}}$.

\textbf{Verifier (normal-quality).}  Softened-tail routing
$w_i^{\text{soft}} = e^{z_i / \tau_r}\big/\sum_j e^{z_j / \tau_r}$, with $\tau_r \to 0$ recovering hard routing.
At each verifier event the primary scores every $k \in \cK_{\text{spec}}$ and accepts iff $\arg\max_v p_t^k(v) = \hat{y}^k$ (probability-threshold and PRISM-gated variants are supported but off by default).
Rejected tokens are remasked and evicted from $\cK_{\text{stable}}$; if the rejection rate exceeds $\eta$, $\cC_{\text{aux}}$ is reset and the verifier is forced again on the corrected sequence.

\subsection{State, Action, and Policy Architecture}

State $s_t = (\bH_t, \bfm_t, \bm{\alpha}_t, \bm{a}_t, \bm{\ell}_t, t/T)$ is built from soft-masked embeddings~\cite{hersche2025softmasked}
\begin{equation}
    \bh_t^k = \lambda(p_t^k)\,\mathbf{E}_{\mask} + (1{-}\lambda(p_t^k))\!\!\sum_{j \in \text{top-}K}\!\! \pi_j^k\,\mathbf{E}_{v_j},
    \quad \lambda(p_t^k) = \omega_s\,\sigma\!\bigl(\omega_a(-H(p_t^k) - \omega_b)\bigr),
    \label{eq:softmask}
\end{equation}
together with the verifier acceptance mask $\alpha_t^k$, normalized cache age $a_t^k$, and last-action one-hot $\ell_t^k$.
The mixing parameters $\omega = (\omega_s, \omega_a, \omega_b)$ are trainable end-to-end (GRPO recomputes $\bh_t$ to flow gradients through $\omega$), but are frozen in the canonical run to isolate the caching policy.

The policy is a 1-layer transformer with $d_\pi{=}128$ (${\sim}$500K parameters) and four Bernoulli heads
\begin{equation*}
    (u_t^k, r_t^k, \kappa_t^k, q_t^k) \sim \prod_{d \in \{u,r,\kappa,q\}}\!\! \text{Ber}\!\bigl(\sigma(f^d(s_t)^k)\bigr),
\end{equation*}
constrained by $u_t^k{=}0$ if unmasked, $r_t^k{=}0$ if masked, and $\kappa_t^k r_t^k{=}0$.
Position $k$ enters $\cK_{\text{stable}}$ only when live, verified, and $\kappa_t^k{=}q_t^k{=}1$, $r_t^k{=}0$.
The canonical GRPO setup freezes $u, r$ (the verifier already authoritative) and trains only $\kappa, q$.

\subsection{Inference: Draft--Validate Loop}

\begin{algorithm}[!ht]
\caption{AOAE Draft-Frontier Inference}
\label{alg:aoae}
\begin{algorithmic}[1]
\State \textbf{Init:} $\by{\leftarrow}(\mask,\ldots,\mask)$, $\cK_{\text{spec}}{\leftarrow}\emptyset$, $\cK_{\text{stable}}{\leftarrow}\emptyset$, $\cC_{\text{aux}}{\leftarrow}\emptyset$
\For{$t = T, \ldots, 1$}
    \If{verifier does not fire (budget: 12 tokens or 4 microsteps)}
        \State Drafter step on $(\bx, \by; \cC_{\text{aux}})$; unmask $u_t$; push tokens to $\cK_{\text{spec}}$
        \State \textbf{continue}
    \EndIf
    \State Run verifier $p_t \leftarrow p_{\btheta}(\cdot \mid \bx, \by)$
    \For{$(k, \hat{y}^k) \in \cK_{\text{spec}}$}
        \State $\alpha_t^k \leftarrow \ind[\arg\max_v p_t^k(v) = \hat{y}^k]$
        \If{$\alpha_t^k = 0$} remask $y^k$; evict $k$ from $\cK_{\text{stable}}$ \EndIf
    \EndFor
    \If{rejection rate $> \eta$} reset $\cC_{\text{aux}}$; re-run verifier \EndIf
    \State Clear $\cK_{\text{spec}}$; build state $s_t$; sample $(u_t, r_t, \kappa_t, q_t)$
    \State Remask $r_t^k{=}1$ positions; evict from $\cK_{\text{stable}}$
    \State Unmask from $\tilde{p}_t = \textsc{Compose}(p_t, d_t, \alpha_t, \gamma)$; commit $\kappa_t{=}1, q_t{=}1, r_t{=}0$ to $\cK_{\text{stable}}$
\EndFor
\State \textbf{Return} $\by$
\end{algorithmic}
\end{algorithm}

The two caches answer different questions: $\cK_{\text{spec}}$ tracks one-step verification debt and is cleared at every verifier event, while $\cK_{\text{stable}}$ marks multi-step persistence and is evicted only by remask or verifier rejection.  Conflating them — as a single accepted-reuse cache does — silently treats unvalidated drafts as persistent state.  The candidate-budget scheduler fires the verifier after 12 accumulated tokens or 4 draft microsteps, with forced first/last events.

\paragraph{Composed prediction.}
\begin{equation}
    \tilde{p}_t^k(v) \propto p_t^k(v) \cdot d_t^k(v)^{\gamma\, \alpha_t^k},
    \label{eq:composed}
\end{equation}
with $\gamma \geq 0$ the composition strength.
At $\gamma{=}0$ this is exactly the verifier distribution.
For numerical stability, the runtime evaluates Eq.~\eqref{eq:composed} in normalized log-probability space — $\log \tilde p \propto \log p_{\text{primary}} + \gamma \alpha_t^k \log d_t^k$ followed by \texttt{logsumexp} renormalization — so the operation is dimensionally consistent at any $\gamma$, and at disagreement positions ($\alpha_t^k{=}0$) returns the verifier logits unchanged.

\subsection{Positional Speculative Caching}

The access head predicts which positions need compute within the next $H$ steps:
\begin{equation}
    \cQ_t = \mathrm{Top}\text{-}B\bigl(\{f^{\text{access}}(s_t)^k\}_k\bigr) \cup \{k : u_t^k = 1 \lor r_t^k = 1\},
\end{equation}
with mandatory positions (those actually changed at step $t$) always included.
$\cQ_t$ gates which entries of $\cK_{\text{stable}}$ are refreshed; only refreshed positions can persist into the next step's stable cache.  This is model-agnostic and applicable to any MDLM exposing per-position uncertainty.

\subsection{GRPO Training}

The base model is frozen.  Compute is measured by an effective-FLOP ratio combining cheap drafter steps (relative cost $c_{\text{aux}}$, count $N_{\text{aux}}$) and the verifier-event miss fraction $m_t$ after $\cK_{\text{stable}}$ reuse:
\begin{equation}
    \mathrm{eff\_flops} = \frac{c_{\text{aux}} N_{\text{aux}} + \sum_{t \in \cV} m_t}{T},
\end{equation}
and the shaped reward is
\begin{equation}
    R = r(\by^*, \hat{\by}) \cdot (1 - \mathrm{eff\_flops})^\alpha
      - \beta\,\overline{\mathrm{Thrash}}
      - w_u f_{\text{unresolved}}
      + w_{F1}\,\overline{F1}_{\text{cache}}
      + w_q F1_{\text{access}},
    \label{eq:reward}
\end{equation}
with $\overline{\mathrm{Thrash}}$ the per-token rate of remasking $\cK_{\text{stable}}$ positions, $f_{\text{unresolved}}$ a penalty for leftover $\mask$, $\overline{F1}_{\text{cache}}$ a soft precision/recall over low-drift cached positions, and $F1_{\text{access}}$ scoring whether $\cQ_t$ anticipates future modifications.  Multiplying correctness by $(1 - \mathrm{eff\_flops})^\alpha$ zeros the reward on wrong answers, preventing speed-gaming~\cite{jazbec2025learning}.

For each prompt we sample $G$ trajectories under $\pi_{\bphi_{\text{old}}}$ and use group-normalized advantage $A^g = (R^g - \bar R)/\hat\sigma_R$:
\begin{equation}
    \cJ(\bphi) = \E \!\left[\, \frac{1}{G}\!\sum_{g=1}^G\! \frac{1}{T{-}\hat T_g}\!\sum_{t=\hat T_g}^{T}\!\min\!\bigl\{\rho_t^g A^g,\,\mathrm{clip}(\rho_t^g, 1{-}\epsilon, 1{+}\epsilon) A^g \bigr\}\right],
\end{equation}
with $\rho_t^g = \pi_{\bphi}(a_t^g\mid s_t^g)/\pi_{\bphi_{\text{old}}}(a_t^g\mid s_t^g)$ and log-ratios clamped before exponentiation.

\section{Experimental Setup}
\label{sec:experiments}
% ======================================================================

\paragraph{Setup.}  LLaDA~2.1-mini (16B MoE, frozen); 1-layer policy ($d_\pi{=}128$, ${\sim}$500K parameters).
Canonical inference: candidate budget 12, max 4 draft microsteps, forced first/last verifier events, argmax-match acceptance, $\gamma{=}0.5$, PRISM off.
$\cK_{\text{stable}}$ is trained and reported; the paper reproduction does not credit stable-KV skipping unless an explicit ablation enables a safe backend path, so reported speed comes from cheap draft microsteps and verifier validation alone.
Hyperparameters: $K{=}5$, $G{=}8$, $\alpha{=}1.0$, $\beta{=}0.1$, $\epsilon{=}0.2$, $c_{\text{aux}}{=}0.35$, $w_{F1}{=}0.02$, $w_q{=}0$; GRPO trains $\kappa, q$ only.

\paragraph{Data and baselines.}  Train: OpenMathInstruct-2 (5K prompts, 3 epochs, 1000 steps).  Eval: GSM8K~\cite{cobbe2021gsm8k} and MATH~\cite{hendrycks2021math}.  Baselines: LLaDA~2.1 speed mode (threshold 0.5, block diffusion), quality mode (threshold 0.7), and Fast-dLLM~\cite{wu2025fastdllm}.

\paragraph{Metrics.}  Accuracy (flexible extraction), throughput (tokens/sec), effective FLOPs, verifier call rate, frontier acceptance/rejection, drafter cache resets, $\cK_{\text{stable}}$ occupancy, and access F1.

\paragraph{Ablations.}
\emph{Speculation:} drafter-only, verifier-only, fixed-step vs.\ candidate-budget scheduling, $\gamma{=}0$ conservative composition.
\emph{Caching:} $\cK_{\text{spec}}$-only, $\cK_{\text{stable}}$-only, combined; stable-KV execution and cache-reset threshold sweeps.
\emph{Policy:} cache/access-only vs.\ full four-head training; without age features; $N_\pi \in \{1,2,4\}$.
