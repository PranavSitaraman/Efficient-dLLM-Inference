\begin{abstract}
% ======================================================================
Masked diffusion language models (MDLMs) generate text by iteratively unmasking tokens in parallel, but most inference methods rely on fixed confidence thresholds or block schedules that impose uniform compute regardless of position difficulty.
We propose \textbf{Any-Order Adaptive Editing (AOAE)}, a self-speculative framework for MoE MDLMs that exploits the natural drafter--verifier split available in a single frozen model under hard versus normal routing.
The hard-routed drafter (${\sim}$1.4B active parameters) accumulates draft tokens cheaply into a transient frontier $\cK_{\text{spec}}$; the normal verifier periodically validates the frontier, accepting matching tokens and remasking rejections.
A composition knob $\gamma$ controls speed--quality trade-offs; setting $\gamma{=}0$ preserves the verifier distribution exactly.
GRPO then trains a lightweight policy (${\sim}$500K parameters) to predict which positions are stable enough to persist in a separate learned cache $\cK_{\text{stable}}$ and which to refresh proactively.
\end{abstract}

% ======================================================================
\section{Background}
\label{sec:background}
% ======================================================================

Let $\bx_0 = (x_0^1, \dots, x_0^L) \in \cV^L$ be a token sequence with mask token $\mask$.
The forward process corrupts each token independently:
\begin{equation}
    q(x_t^k \mid x_0^k) = \begin{cases} 1{-}t, & x_t^k = x_0^k, \\ t, & x_t^k = \mask, \end{cases}
    \quad t \in [0,1].
    \label{eq:forward}
\end{equation}
A denoising network $p_{\btheta}$ trained via the masked ELBO~\cite{ou2024your,sahoo2024simple} starts from a fully masked response $\by_T = (\mask,\ldots,\mask)$ and iteratively selects positions $\cU_t \subseteq \cM_t$ to unmask.
The choice of $\cU_t$ governs both speed and quality~\cite{jazbec2025learning,wu2025fastdllm}.

Existing acceleration work falls into three categories.
\emph{Adaptive unmasking} learns better $\cU_t$ selections~\cite{jazbec2025learning,kim2025trainworst}.
\emph{KV-cache reuse} freezes stable key-value states across steps~\cite{ma2025dkv,elastic_cache}, but constrains decoding to block or prefix patterns.
\emph{Self-correction} reverts low-confidence tokens via a learned or threshold policy~\cite{kim2025prism}.
AOAE integrates all three ideas into a single trainable framework.

\section{Method}
\label{sec:method}
% ======================================================================

\subsection{Dual-Model MoE Architecture}

LLaDA~2.1-mini~\cite{llada21} is a 16B-parameter MoE model with $E{=}128$ routed experts; $k{=}8$ are activated per token under hard top-$k$ routing (${\sim}$1.4B active).
AOAE uses two operational modes of the \emph{same} frozen model:

\textbf{Drafter (auxiliary).}  Hard-routed, 8$\times$ cheaper per token.  Runs an aggressive inference shortcut (S-mode thresholding, prefix-cache reuse).  Each unmasked token is appended to the transient frontier $\cK_{\text{spec}}$.

\textbf{Verifier (primary).}  Runs the normal mode with optional soft routing:
\begin{equation}
    w_i^{\text{soft}} = \frac{\exp(z_i / \tau_r)}{\sum_j \exp(z_j / \tau_r)}, \quad \tau_r \to 0 \text{ recovers hard routing.}
\end{equation}
Both modes share frozen weights; routing is toggled in-place.
At each verifier event, the primary scores every position in $\cK_{\text{spec}}$: a frontier token is accepted iff $\arg\max_v p_t^k(v) = \hat{y}^k$ (configurable to a probability threshold).  Rejected tokens are remasked and evicted from $\cK_{\text{stable}}$; if the frontier rejection rate exceeds $\eta$, the drafter cache $\cC_{\text{aux}}$ is reset.

\subsection{State, Action, and Policy Architecture}

The policy state uses \emph{soft-masked embeddings}~\cite{hersche2025softmasked}:
\begin{equation}
    \bh_t^k = \lambda(p_t^k)\,\mathbf{E}_{\mask} + (1{-}\lambda(p_t^k))\sum_{j \in \text{top-}K} \pi_j^k\,\mathbf{E}_{v_j},
    \quad \lambda(p_t^k) = \omega_s \cdot \sigma\!\bigl(\omega_a(-H(p_t^k) - \omega_b)\bigr),
    \label{eq:softmask}
\end{equation}
with learnable $\omega = (\omega_s, \omega_a, \omega_b)$ trained jointly via GRPO.
The state is $s_t = (\bH_t, \bfm_t, \bm{\alpha}_t, \bm{a}_t, \bm{\ell}_t, t/T)$, where $\alpha_t^k$ is the verifier's acceptance signal, $a_t^k$ is a normalized cache age, and $\ell_t^k$ encodes the last action.

The policy has four Bernoulli heads; the canonical GRPO configuration freezes unmask/remask and trains only cache/access:
\begin{align}
    u_t^k &\sim \text{Ber}(\sigma(f^{\text{unmask}}(s_t)^k)), &
    r_t^k &\sim \text{Ber}(\sigma(f^{\text{remask}}(s_t)^k)), \\
    \kappa_t^k &\sim \text{Ber}(\sigma(f^{\text{cache}}(s_t)^k)), &
    q_t^k &\sim \text{Ber}(\sigma(f^{\text{access}}(s_t)^k)).
\end{align}
Validity constraints: $u_t^k = 0$ if already unmasked; $r_t^k = 0$ if masked; $\kappa_t^k \cdot r_t^k = 0$.
Position $k$ enters $\cK_{\text{stable}}$ only when $\kappa_t^k {=} 1$, $q_t^k {=} 1$, $r_t^k {=} 0$, and $y^k \neq \mask$.

The policy is a 1-layer transformer with $d_\pi{=}128$ (${\sim}$500K parameters).

\subsection{Inference: Draft--Validate Loop}

\begin{algorithm}[!ht]
\caption{AOAE Draft-Frontier Inference}
\label{alg:aoae}
\begin{algorithmic}[1]
\State \textbf{Init:} $\by{\leftarrow}(\mask,\ldots,\mask)$, $\cK_{\text{spec}}{\leftarrow}\emptyset$, $\cK_{\text{stable}}{\leftarrow}\emptyset$, $\cC_{\text{aux}}{\leftarrow}\emptyset$
\For{$t = T, \ldots, 1$}
    \If{verifier does not fire (budget: 12 tokens or 4 microsteps)}
        \State Drafter step on $(\bx, \by; \cC_{\text{aux}})$; unmask $u_t$; push tokens to $\cK_{\text{spec}}$
        \State \textbf{continue}
    \EndIf
    \State Run verifier $p_t \leftarrow p_{\btheta}(\cdot \mid \bx, \by)$
    \For{$(k, \hat{y}^k) \in \cK_{\text{spec}}$}
        \State $\alpha_t^k \leftarrow \ind[\arg\max_v p_t^k(v) = \hat{y}^k]$
        \If{$\alpha_t^k = 0$} remask $y^k$; evict $k$ from $\cK_{\text{stable}}$ \EndIf
    \EndFor
    \If{rejection rate $> \eta$} reset $\cC_{\text{aux}}$; re-run verifier \EndIf
    \State Clear $\cK_{\text{spec}}$; build state $s_t$; sample $(u_t, r_t, \kappa_t, q_t)$
    \State Remask $r_t^k{=}1$ positions; evict from $\cK_{\text{stable}}$
    \State Unmask from $\tilde{p}_t = \textsc{Compose}(p_t, d_t, \alpha_t, \gamma)$; commit $\kappa_t{=}1, q_t{=}1, r_t{=}0$ to $\cK_{\text{stable}}$
\EndFor
\State \textbf{Return} $\by$
\end{algorithmic}
\end{algorithm}

$\cK_{\text{spec}}$ is \emph{transient}: it tracks unvalidated draft work and is cleared at every verifier event.  $\cK_{\text{stable}}$ is \emph{persistent}: it survives across steps and is evicted only by remask or verifier rejection.  The canonical scheduler fires the verifier after 12 accumulated candidate tokens or 4 draft microsteps, with forced first and last events.

\paragraph{Composed prediction.}
\begin{equation}
    \tilde{p}_t^k(v) \propto p_t^k(v) \cdot d_t^k(v)^{\gamma \cdot \alpha_t^k},
    \label{eq:composed}
\end{equation}
where $\gamma \geq 0$ is the composition strength and $\alpha_t^k$ is the acceptance mask.
At $\gamma{=}0$ the distribution equals the verifier exactly; at $\gamma{>}0$ it concentrates on tokens both models agree on.

\subsection{Positional Speculative Caching}

The access head predicts which positions will need compute within the next $H$ steps.  The executed access set is:
\begin{equation}
    \cQ_t = \mathrm{Top}\text{-}B\!\bigl(\{f^{\text{access}}(s_t)^k\}_{k}\bigr) \cup \{k : u_t^k{=}1 \lor r_t^k{=}1\},
\end{equation}
where mandatory positions (those actually changed at step $t$) are always included.  This is model-agnostic and compatible with any MDLM that exposes per-position uncertainty.

\subsection{GRPO Training}

The base model is frozen.  Let $c_{\text{aux}}$ be the relative cost of one drafter step and $m_t$ the verifier miss fraction after stable-cache reuse:
\begin{equation}
    \mathrm{eff\_flops} = \frac{c_{\text{aux}} N_{\text{aux}} + \sum_{t \in \cV} m_t}{T}.
\end{equation}
The reward combines task correctness, speed, and auxiliary penalties:
\begin{equation}
    R = r(\by^*, \hat{\by}) \cdot (1 - \mathrm{eff\_flops})^\alpha
      - \beta\,\overline{\mathrm{Thrash}}
      - w_u\,f_{\text{unresolved}}
      + w_{F1}\,\overline{F1}_{\text{cache}}
      + w_q\,F1_{\text{access}},
    \label{eq:reward}
\end{equation}
where $\overline{\mathrm{Thrash}}$ is the per-token rate of remasking stable-cached positions; $f_{\text{unresolved}}$ penalizes leftover $\mask$ tokens; $\overline{F1}_{\text{cache}}$ rewards $\kappa_t$ for committing low-drift positions; and $F1_{\text{access}}$ measures whether $\cQ_t$ anticipates future modifications.  The multiplicative speed term ensures incorrect answers receive zero reward, preventing speed-gaming~\cite{jazbec2025learning}.

For each prompt, $G$ trajectories are sampled under $\pi_{\bphi_{\text{old}}}$; with advantage $A^g = R^g - \bar{R}$:
\begin{equation}
    \cJ(\bphi) = \E \left[ \frac{1}{G} \sum_{g=1}^G \frac{1}{T{-}\hat{T}_g} \sum_{t=\hat{T}_g}^{T} \min\!\Big\{ \rho_t^g A^g,\; \mathrm{clip}(\rho_t^g, 1{-}\epsilon, 1{+}\epsilon) A^g \Big\} \right],
\end{equation}
where $\rho_t^g = \pi_{\bphi}(a_t^g \mid s_t^g) / \pi_{\bphi_{\text{old}}}(a_t^g \mid s_t^g)$ and log-ratios are clamped before exponentiation.

\section{Experimental Setup}
\label{sec:experiments}
% ======================================================================

\paragraph{Model and policy.}  LLaDA~2.1-mini (16B MoE, frozen).  Policy: 1-layer transformer, $d_\pi{=}128$, ${\sim}$500K parameters.

\paragraph{Canonical configuration.}  Candidate budget 12, max 4 draft microsteps, forced first/last verifier events, argmax-match acceptance.  Hyperparameters: $K{=}5$, $G{=}8$, $\alpha{=}1.0$, $\beta{=}0.1$, $\epsilon{=}0.2$, $\gamma{=}0.5$, $c_{\text{aux}}{=}0.35$.  GRPO trains cache/access heads only.

\paragraph{Data.}  GSM8K~\cite{cobbe2021gsm8k} and MATH~\cite{hendrycks2021math} for evaluation; OpenMathInstruct-2 for policy training (50K prompts, 3 epochs, 2000 steps).

\paragraph{Baselines.}  LLaDA~2.1 speed mode (threshold 0.5, block diffusion), quality mode (threshold 0.7), and Fast-dLLM~\cite{wu2025fastdllm}.

\paragraph{Metrics.}  Accuracy (flexible extraction for masked-diffusion output), throughput (tokens/sec), effective FLOPs, verifier call rate, draft microsteps per step, frontier acceptance/rejection rates, drafter cache resets, stable-cache occupancy, access F1, and refresh ratio.

\paragraph{Ablations.}
\begin{itemize}[leftmargin=*,itemsep=1pt]
    \item \textbf{Speculation}: drafter-only, verifier-only, fixed-step vs.\ candidate-budget scheduling, $\gamma{=}0$ conservative composition.
    \item \textbf{Caching}: $\cK_{\text{spec}}$ only, $\cK_{\text{stable}}$ only, combined; cache-reset threshold sweep.
    \item \textbf{Policy}: cache/access-only vs.\ full four-head training; no age feature; $N_\pi \in \{1,2,4\}$.
\end{itemize}
